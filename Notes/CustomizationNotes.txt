------------------------ SOURCE CUSTOMIZATION ---------------------------

M src/main.c
-- enable sqlite3_trace when SQLITE_AUTO_TRACE is defined

M src/mem1.c
-- use custom malloc zone on single core systems

M src/os_unix.c
-- hostid via gethostuuid
-- read-only file system support for proxy locking
-- improved handling of simulated shared lock on AFP/SMB
-- 64-bit AFP customization
-- proxy lock breaking & recreating local lock files
-- support SQLITE_OPEN_AUTOPROXY flag

M src/pcache1.c
-- disabled if SQLITE_OMIT_PCACHE1_PCACHE

A src/pcache_purgeable.c
-- purgeable pcache implementation 

M src/prepare.c
-- log statement prepares when SQLITE_AUTO_TRACE > 1

M src/sqlite.h.in
-- added SQLITE_OPEN_AUTOPROXY

M src/sqliteInt.h
-- alternative SQLITE_INT_TO_PTR defs for llvm compatibility
-- support for SQLITE_ENABLE_PURGEABLE_PCACHE macro

M src/vdbeapi.c
-- checking for NULL statement for <rdar://6646331> 

M src/vdbe.c
-- http://www.sqlite.org/cvstrac/tktview?tn=3774

M src/walker.c
-- http://www.sqlite.org/cvstrac/tktview?tn=3791
